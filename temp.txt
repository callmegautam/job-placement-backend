const isEmailOrUsernameTaken = async (email: string, username: string, excludeId?: number) => {
    return await db
        .select()
        .from(student)
        .where(
            or(
                and(eq(student.email, email), sql`${student.id} != ${excludeId}`),
                and(eq(student.username, username), sql`${student.id} != ${excludeId}`)
            )
        );
};
